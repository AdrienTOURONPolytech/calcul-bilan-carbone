    <script>

        function calculateCarbon() {

            const snapsPhoto = parseInt(document.getElementById('snapsPhoto').value) || 0;

            const snapsVideo = parseInt(document.getElementById('snapsVideo').value) || 0;

            const instagramHours = parseFloat(document.getElementById('instagramHours').value) || 0;

            const network = document.getElementById('network').value;


            let networkFactor = 1;

            if (network === '4g') networkFactor = 2.5;

            if (network === '5g') networkFactor = 3.0;


            const photoCO2 = 0.35;

            const videoCO2 = 2.0;

            const instaCO2PerHour = 100;


            const snapsCO2 = (snapsPhoto * photoCO2 + snapsVideo * videoCO2);

            const instaCO2 = instagramHours * instaCO2PerHour;


            const totalCO2 = (snapsCO2 + instaCO2) * networkFactor;


            document.getElementById('result').innerHTML = 

                `Votre bilan carbone quotidien estimé est : <strong>${totalCO2.toFixed(1)} g CO₂e</strong><br>

                 (≈ ${(totalCO2/1000).toFixed(3)} kg CO₂e par jour)`;

        }

    </script>
